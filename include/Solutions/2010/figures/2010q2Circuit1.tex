\documentclass[crop,tikz]{standalone}

\usepackage{../../../../src/tikzappmath}

\usetikzlibrary{scopes}
\usetikzlibrary{arrows,shapes.gates.logic.US,shapes.gates.logic.IEC,calc}


\begin{document}
	\begin{tikzpicture}
		

		\def\xspace{3} % set the horizonal distance between gates
		\def\yspace{2} % set the vertical distance between gates
		\def\shift{0.5cm} % sets some spacing to draw connecting lines
		
		% Draw gates at nodes
		% define first col of gates
		\node[not gate US,gatestyle] at (0,0) (x1) {};

		% define second col of gates
		\node[or gate US, gatestyle] at (3,-1) (x2) {};\


		% Connect all gates internally
		\draw (x1.output) -| ([xshift = -\shift]x2.input 1) --++(\shift,0);

		% draw end piece
		\draw (x2.output) --++ (1.5,0);

		% inputs 1
		\draw (x1.input) -|++ (-\shift,\shift) --++(-\shift,0) node[left,]{$a$};
		\draw (x2.input 2) -|++ (-\shift,-\shift) --++(-\shift,0) node[left]{$b$};

		% Insert labels
		\node[above] at ($(x1.output) + (\shift,0)$) (l1) {\gatetext$\sim a$};
		\node[above] at ($(x2.output) + (\shift,0)$) (l1) {\gatetext$\sim a \lor b$};

	\end{tikzpicture}
	
\end{document}
